use cursive::views::TextView;
use cursive::Cursive;

fn main() {
    let mut win = cursive::default();

    // We can quit by pressing `q`
    win.add_global_callback('q', Cursive::quit);

    // Add a simple view
    win.add_layer(TextView::new(
        "Hello Curses!\n\
         Press q to quit.",
    ));

    // Run the event loop
    win.run();
}
